package server;

import com.squareup.moshi.*;
import java.lang.reflect.Type;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import spark.Request;
import spark.Response;
import spark.Route;

public class HelpHandler implements Route {

  /**
   * handle a new request to this endpoint!
   *
   * @param request the request to handle (including query params)
   * @param response use to modify properties of the response
   * @return response content
   */
  @Override
  public Object handle(Request request, Response response) {

    // grab request url
    String url = request.url();
    return new HelpResponse(url, "success", request.queryParams("endpoint")).serialize();
  }

  /**
   * Makes a response map to send, containing details about the request, and our response
   *
   * @param url - the url of the request we received
   * @param response - error/success plus details
   */
  public record HelpResponse(String url, String response, String endpoint) {

    /**
     * @return url, result, and help information, serialized as Json
     */
    String serialize() {
      try {
        // add to our response map!
        HashMap<String, Object> map = new HashMap<>();
        map.put("request", url);
        map.put("result", response);
        map.put("endpoint specified for help", endpoint);

        // serve help!
        String helpString =
            "this API is designed to load, view, and search CSV "
                + "files, as well as serve weather information. Responses are "
                + "encoded as Json data, and every response from this API will "
                + "contain the \"request\" and \"result\" fields. It will also "
                + "contain the values of the required parameters for the endpoint "
                + "you requested, even if no value was provided. If you'd like "
                + "information on a certain endpoint, include a parameter in "
                + "search named \"endpoint\" with the value being the name of one "
                + "of the valid endpoints";
        if (endpoint != null) {
          switch (endpoint) {
            case "loadcsv" -> helpString =
                "The \"loadcsv\" endpoint accepts exactly "
                    + "2 query parameters: \"filepath\" and \"hasHeaders\". This "
                    + "endpoint loads a csv file located at filepath for viewing "
                    + "and/or searching. filepath must start with \"/data\", end with "
                    + "\".csv\", and contain only one dot '.' (the one in \".csv\"). "
                    + "Uploading files is not supported. Loading a new file when one "
                    + "is already loaded will replace the previously loaded file.";
            case "viewcsv" -> helpString =
                "The \"viewcsv\" endpoint accepts exactly "
                    + "0 query parameters. It outputs the contents of the currently "
                    + "loaded csv file. A file must be loaded via request to "
                    + "\"loadcsv\" in order to be viewed";
            case "searchcsv" -> helpString =
                "The \"searchcsv\" endpoint accepts exactly "
                    + "3 query parameters: \"searchValue\", \"identifier\", and "
                    + "\"index\". This endpoint requires a file to have been loaded "
                    + "via the \"loadcsv\" endpoint. This endpoint searches the "
                    + "loaded file for the value assigned to \"searchValue\", "
                    + "searching in the column whose header is the value of "
                    + "\"identifier\" when \"index\" == -2, or all columns when "
                    + "\"index\" == -1, or the column specified by \"index\" when "
                    + "\"index\" > 0. The entire row is returned when a matching "
                    + "value is found, and nothing is returned otherwise. Note that you "
                    + "may not search in a column specified by \"identifier\" if when "
                    + "you requested the \"loadcsv\" endpoint, you did not include "
                    + "\"hasHeaders=true\". Similarly, you may not use \"index\" == "
                    + "-2 in this case. You must provide \"identifier=null\", and for "
                    + "\"index\" you must provide a number greater than -2. If "
                    + "\"index\" is out of bounds, an error will be thrown.";
            case "weather" -> helpString =
                "The \"weather\" endpoint accepts exactly "
                    + "2 parameters, longitude and latitude. longitude must be less than 180 "
                    + "and more than -180, and latitude must be less than 90 and greater than -90. "
                    + "This endpoint will call the NWS API to return current temperature, and its "
                    + "unit. This endpoint can returned cached previous results if the lat and long "
                    + "of the request are close enough to a previous one (a metric set by the "
                    + "developer when starting the server), and the cache will be automatically "
                    + "evicted after an amount of time also set by the developer. This endpoint "
                    + "will indicate whether the response data was pulled from the api or a new "
                    + "api request.";
          }
        }
        map.put("help information:", helpString);
        map.put("valid endpoints", List.of("help", "loadcsv", "viewcsv", "searchcsv", "weather"));

        // .toJson our response map!
        Type responseMap = Types.newParameterizedType(Map.class, String.class, Object.class);
        Moshi moshi = new Moshi.Builder().build();
        JsonAdapter<Map<String, Object>> adapter = moshi.adapter(responseMap);

        return adapter.toJson(map);
      } catch (Exception e) {
        // For debugging purposes, show in the console _why_ this fails
        // Otherwise we'll just get an error 500 from the API in integration
        // testing.
        e.printStackTrace();
        throw e;
      }
    }
  }
}
