package server;

import com.squareup.moshi.JsonAdapter;
import com.squareup.moshi.Moshi;
import com.squareup.moshi.Types;
import java.lang.reflect.Type;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Objects;

import server.Server.CSVSearcherContainer;
import spark.Request;
import spark.Response;
import spark.Route;

public class SearchCSVHandler implements Route {

  private CSVSearcherContainer searcherContainer;

  /**
   * Constructor accepts some shared state
   *
   * @param searcherContainer the shared searcherContainer object that all endpoints have access to
   *                          (note: we *DON'T* want to make a defensive copy here!)
   */
  public SearchCSVHandler(CSVSearcherContainer searcherContainer) {
    this.searcherContainer = searcherContainer;
  }

  /**
   *Handles the current searchCSV request
   * @return SearchCSV Response object based on the current searchCSV request
   */
  @Override
  public Object handle(Request request, Response response) {
    // grab input params
    String searchValue = request.queryParams("searchValue");
    String identifier = request.queryParams("identifier");
    String index = request.queryParams("index");

    // grab request url
    String url = request.url();

    // error check params - we should have exactly 3
    if (request.queryParams().size() != 3) {
      return new SearchCSVResponse(url, searchValue, identifier, index, null,
          "error_bad_request: Not exactly 3 params, should be searchValue, and identifier"
              + " (header name or null), and index (-2 if using header, -1 to " +
            "search all columns, or a number greater than 0 for a specific " +
            "column index to search).").serialize();
    }

    // error check params - we should have searchValue, identifier, and index
    if (searchValue == null || identifier == null || index == null) {
      return new SearchCSVResponse(url, searchValue, identifier, index, null,
        "error_bad_request: No values given for all required parameters, " +
          "should be searchValue, and identifier (header name or null), and " +
          "index (-2 if using header, -1 to search all columns, or a number " +
          "greater than 0 for a specific column index to search).").serialize();
    }

    // error check index
    int parsedIndex;
    try {
      parsedIndex = Integer.parseInt(index);
    } catch (NumberFormatException nfException) {
      // not a number after -index
      return new SearchCSVResponse(url, searchValue, identifier, index, null,
        "error_bad_request: Value of index was not a number - must be " +
          "an integer between -2 and the number of columns in the loaded file, " +
          "inclusive.").serialize();
    }

    // check for null identifier
    if(identifier.equals("null")){
      identifier = null;
    }

    // check if file has been loaded
    if (searcherContainer.searcher == null) {
      return new SearchCSVResponse(url, searchValue, identifier, index, null,
        "error_datasource: No CSV loaded").serialize();
    }

    // now search and return result (if no exceptions)
    try {
      List<List<String>> csvData = searcherContainer.searcher.searchFile(searchValue, identifier, parsedIndex);
      /** fixing bug ?*/
      if (csvData.isEmpty()) {
        return new SearchCSVResponse(url, searchValue, identifier, index, csvData, "failure").serialize();
      }
      /** */
      return new SearchCSVResponse(url, searchValue, identifier, index, csvData, "success").serialize();
    } catch (Exception ex) {
      return new SearchCSVResponse(url, searchValue, identifier, index, null, "error_bad_request: " + ex.getMessage()).serialize();
    }
  }

  /**
   * Makes a response map to send, containing details about the request, and our response
   *
   * @param url         - the url of the request we received
   * @param searchValue - the value of the searchValue query param
   * @param identifier  - the value of the identifier query param
   * @param index       - the value of the index query param
   * @param response    - error/success plus details
   * @param csvData     - the value of the CSVData we searched for
   */
  public record SearchCSVResponse(String url, String searchValue, String identifier, String index,
                                  List<List<String>> csvData, String response) {

    /**
     * @return url, filepath, hasHeaders, and response, serialized as Json
     */
    String serialize() {
      try {
        // add to our response map!
        HashMap<String, Object> map = new HashMap<>();
        map.put("request", url);
        map.put("searchValue", Objects.requireNonNullElse(searchValue, "null"));
        map.put("identifier", Objects.requireNonNullElse(identifier, "null"));
        map.put("index", Objects.requireNonNullElse(index, "null"));
        map.put("result", response);
        map.put("data", Objects.requireNonNullElse(csvData, "null"));

        // .toJson our response map!
        Type responseMap = Types.newParameterizedType(Map.class, String.class, Object.class);
        Moshi moshi = new Moshi.Builder().build();
        JsonAdapter<Map<String, Object>> adapter = moshi.adapter(responseMap);

        return adapter.toJson(map);
      } catch (Exception e) {
        // For debugging purposes, show in the console _why_ this fails
        // Otherwise we'll just get an error 500 from the API in integration
        // testing.
        e.printStackTrace();
        throw e;
      }
    }
  }
}
