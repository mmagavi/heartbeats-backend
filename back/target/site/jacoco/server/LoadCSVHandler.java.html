package server;

import CSVUtilities.CSVParser.FactoryFailureException;
import CSVUtilities.CSVSearcher.CSVSearcher;
import com.squareup.moshi.JsonAdapter;
import com.squareup.moshi.Moshi;
import com.squareup.moshi.Types;
import java.io.IOException;
import java.lang.reflect.Type;
import java.util.HashMap;
import java.util.Map;
import java.util.Objects;
import server.Server.CSVSearcherContainer;
import spark.Request;
import spark.Response;
import spark.Route;

/**
 * Handler class for the loadcsv endpoint.
 *
 * <p>This endpoint takes a basic GET request with no Json body, and returns a Json object in
 * response.
 */
public class LoadCSVHandler implements Route {
  protected final CSVSearcherContainer searcherContainer;

  /**
   * Constructor accepts some shared state
   *
   * @param searcher the shared searcher object that all endpoints have access to (note: we *DON'T*
   *     want to make a defensive copy here!)
   */
  public LoadCSVHandler(CSVSearcherContainer searcher) {
    this.searcherContainer = searcher;
  }

  /**
   * handle a new request to this endpoint!
   *
   * @param request the request to handle (including query params)
   * @param response use to modify properties of the response
   * @return response content
   */
  @Override
  public Object handle(Request request, Response response) {
    // grab input params
    String filepath = request.queryParams("filepath");
    String hasHeaders = request.queryParams("hasHeaders");

    // grab request url
    String url = request.url();

    // error check params - we should have exactly 2
    if (request.queryParams().size() != 2) {
      return new LoadCSVResponse(
              url,
              filepath,
              hasHeaders,
              "error_bad_json: Not exactly 2 params, should be "
                  + "filepath and hasHeaders (true or false)")
          .serialize();
    }

    // error check params - we should have filepath and hasHeaders
    if (filepath == null || hasHeaders == null) {
      return new LoadCSVResponse(
              url,
              filepath,
              hasHeaders,
              "error_bad_json: Values not given for all parameters, "
                  + "should be filepath and hasHeaders (true or false)")
          .serialize();
    }

    // hasHeadersBool (from hasHeaders param)
    boolean hasHeadersBool;
    if (hasHeaders.equalsIgnoreCase("true")) {
      hasHeadersBool = true;
    } else if (hasHeaders.equalsIgnoreCase("false")) {
      hasHeadersBool = false;
    } else {
      return new LoadCSVResponse(
              url, filepath, hasHeaders, "error_bad_json: hasHeaders must be true or false")
          .serialize();
    }

    // sanitize and fix filepath input!
    String cleanFilepath = fixFilepath(filepath);
    if (cleanFilepath == null) {
      return new LoadCSVResponse(
              url,
              filepath,
              hasHeaders,
              "error_datasource: "
                  + filepath
                  + " is not a proper file"
                  + "path. filepath must start with \"/data\" or \"/back/data\", end with \".csv\", "
                  + "and contain only one dot '.' (the one in \".csv\")")
          .serialize();
    }
    CSVSearcher searcher;

    // now make new CSVSearcher and return it (if no exceptions)
    try {
      searcher = new CSVSearcher(cleanFilepath, hasHeadersBool);
      // set our fields appropriately and return!
      searcherContainer.searcher = searcher;
      return new LoadCSVResponse(url, filepath, hasHeaders, "success").serialize();
    } catch (IOException ex) {
      // if exception, print our exception to output and return failure!
      ex.printStackTrace();
      return new LoadCSVResponse(
              url, filepath, hasHeaders, "error_datasource: " + filepath + " could not be opened")
          .serialize();
    } catch (FactoryFailureException ex) {
      ex.printStackTrace();
      return new LoadCSVResponse(
              url,
              filepath,
              hasHeaders,
              "error_bad_request: FactoryFailureException on backend "
                  + "during CSVSearcher creation")
          .serialize();
    } catch (Exception ex) {
      ex.printStackTrace();
      return new LoadCSVResponse(
              url,
              filepath,
              hasHeaders,
              "error_bad_request: Exception on backend during " + "CSVSearcher creation")
          .serialize();
    }
  }

  /**
   * A method to fix and sanitize our input filepath!
   *
   * @param filepath - the raw, possibly dirty filepath
   * @return the clean filepath (or null, indicating bad filepath)
   */
  private String fixFilepath(String filepath) {
    // sanitize our filepath input!
    int len = filepath.length();
    // verify that last four characters are ".csv"
//    if ((len < 5)
//        || (filepath.charAt(len - 4) != '.')
//        || (filepath.charAt(len - 3) != 'c')
//        || (filepath.charAt(len - 2) != 's')
//        || (filepath.charAt(len - 1) != 'v')) {
//      return null;
//    }

    // verify correct number of dots!
    int dotCount = 0;
    for (int i = 0; i < filepath.length(); i++) {
      if (filepath.charAt(i) == '.') {
        if (dotCount > 0) {
          return null;
        }
        dotCount++;
      }
    }

    String[] tokens = filepath.split("/");
    // check valid windows machine path
    // token before first slash = "", token after first slash = "data"
    Boolean badWindowsPath = tokens.length < 2 || (!tokens[0].equals("")) || (!tokens[1].equals("data"));
    // check valid mac machine path
    // token before first slash = "", token after first slash = "back", token
    // after second slash = "data"
    Boolean badMacPath = tokens.length < 3 || (!tokens[0].equals("")) || (!tokens[1].equals("backend")) || (!tokens[2].equals("data"));
    if (badWindowsPath && badMacPath) {
      return null;
    }

    // add the project directory to the start of filepath
    // so if filepath == "/data/mockCSV/birthday.csv" we have the absolute
    // filepath of our file!
    return System.getProperty("user.dir") + filepath;
  }

  /**
   * Makes a response map to send, containing details about the request, and our response
   *
   * @param url - the url of the request we received
   * @param filepath - the value of the filepath query param
   * @param hasHeaders - the value of the hasHeaders query param
   * @param response - error/success plus details
   */
  public record LoadCSVResponse(String url, String filepath, String hasHeaders, String response) {
    /**
     * @return url, filepath, hasHeaders, and response, serialized as Json
     */
    String serialize() {
      try {
        // add to our response map!
        HashMap<String, Object> map = new HashMap<>();
        map.put("request", url);
        map.put("hasHeaders", Objects.requireNonNullElse(hasHeaders, "null"));
        map.put("filepath", Objects.requireNonNullElse(filepath, "null"));
        map.put("result", response);

        // .toJson our response map!
        Type responseMap = Types.newParameterizedType(Map.class, String.class, Object.class);
        Moshi moshi = new Moshi.Builder().build();
        JsonAdapter<Map<String, Object>> adapter = moshi.adapter(responseMap);
        return adapter.indent("    ").toJson(map);

      } catch (Exception e) {
        // For debugging purposes, show in the console _why_ this fails
        // Otherwise we'll just get an error 500 from the API in integration
        // testing.
        e.printStackTrace();
        throw e;
      }
    }
  }
}
