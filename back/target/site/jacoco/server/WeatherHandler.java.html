package server;

import com.squareup.moshi.JsonAdapter;
import com.squareup.moshi.Moshi;
import com.squareup.moshi.Types;
import java.lang.reflect.Type;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Objects;
import java.util.concurrent.ExecutionException;
import spark.Request;
import spark.Response;
import spark.Route;

public class WeatherHandler implements Route {

  // class that helps with caching!
  WeatherResponseCache weatherResponseCache;

  /**
   * Constructor that takes in developer options: cache eviction time and cache "close enough"
   * distance
   *
   * @param maxSize - max size of cache
   * @param evictionTime - cache eviction time (int in minutes)
   * @param maxDistance - maximum distance which can be considered "close enough" for cached results
   *     to be returned (int in miles)
   */
  public WeatherHandler(int maxSize, int evictionTime, int maxDistance, Loader myLoader) {
    this.weatherResponseCache =
        new WeatherResponseCache(maxSize, evictionTime, maxDistance, myLoader);
  }

  /**
   * Handle the request to the weather endpoint
   *
   * @return Weather Response object related to the current request
   * @throws NumberFormatException if the given longitude or latitude are not valid or cannot be
   *     converted
   */
  @Override
  public Object handle(Request request, Response response) throws NumberFormatException {
    // grab input params
    String latitude = request.queryParams("latitude");
    String longitude = request.queryParams("longitude");

    // grab request url
    String url = request.url();

    // error check params - we should have exactly 2
    if (request.queryParams().size() != 2) {
      return new WeatherResponse(
              url,
              latitude,
              longitude,
              null,
              null,
              null,
              "error_bad_request: Not exactly 2 params, should be " + "latitude and longitude")
          .serialize();
    }

    // check that lat and long actually exist
    if (latitude.equals("") || longitude.equals("")) {
      return new WeatherResponse(
              url,
              latitude,
              longitude,
              null,
              null,
              null,
              "error_bad_request: No values given for all required parameters, "
                  + "should be latitude and longitude.")
          .serialize();
    }

    // check that lat and long are actual numbers
    float latitudeNum = -1;
    float longitudeNum = -1;
    try {
      latitudeNum = Float.parseFloat(latitude);
      longitudeNum = Float.parseFloat(longitude);
      if (longitudeNum < -180 || longitudeNum > 180 || latitudeNum < -90 || latitudeNum > 90) {
        throw new NumberFormatException();
      }
    } catch (NumberFormatException nfx) {
      return new WeatherResponse(
              url,
              latitude,
              longitude,
              null,
              null,
              null,
              "error_bad_request: longitude must be less than 180 and more than "
                  + "-180, and latitude must be less than 90 and greater than -90.")
          .serialize();
    }

    // ask our cache to grab temperature and unit data for this lat and long!
    List<String> cacheRet;
    try {
      cacheRet = weatherResponseCache.checkCache(latitudeNum, longitudeNum);
    } catch (ExecutionException e) {
      System.out.println(e.getLocalizedMessage());
      e.printStackTrace();
      return new WeatherResponse(
              url,
              latitude,
              longitude,
              null,
              null,
              null,
              "error_bad_request: NWS API returned response code: 404")
          .serialize();
    } catch (Exception e) {
      // For debugging purposes, show in the console _why_ this fails
      // Otherwise we'll just get an error 500 from the API in integration
      // testing.
      e.printStackTrace();
      throw e;
    }
    // make a new response and serialize it!
    return new WeatherResponse(
            url, latitude, longitude, cacheRet.get(0), cacheRet.get(1), cacheRet.get(2), "success")
        .serialize();
  }

  /**
   * Makes a response map to send, containing details about the request, and our response
   *
   * @param url - the url of the request we received
   * @param latitude - the value of the filepath query param
   * @param longitude - the value of the hasHeaders query param
   * @param temperature - the current temperature
   * @param unit - unit for our temperature reading
   * @param source - the source of this response (NWS api or cache?)
   * @param response - error/success plus details
   */
  public record WeatherResponse(
      String url,
      String latitude,
      String longitude,
      String temperature,
      String unit,
      String source,
      String response) {
    /**
     * @return url, filepath, hasHeaders, and response, serialized as Json
     */
    String serialize() {
      try {
        // add to our response map!
        HashMap<String, Object> map = new HashMap<>();
        map.put("request", url);
        map.put("latitude", Objects.requireNonNullElse(latitude, "null"));
        map.put("longitude", Objects.requireNonNullElse(longitude, "null"));
        map.put("result", response);
        map.put("source", Objects.requireNonNullElse(source, "null"));
        map.put("temperature", Objects.requireNonNullElse(temperature, "null"));
        map.put("unit", Objects.requireNonNullElse(unit, "null"));

        // .toJson our response map!
        Type responseMap = Types.newParameterizedType(Map.class, String.class, Object.class);
        Moshi moshi = new Moshi.Builder().build();
        JsonAdapter<Map<String, Object>> adapter = moshi.adapter(responseMap);

        return adapter.toJson(map);
      } catch (Exception e) {
        // For debugging purposes, show in the console _why_ this fails
        // Otherwise we'll just get an error 500 from the API in integration
        // testing.
        e.printStackTrace();
        throw e;
      }
    }
  }
}
