package server;

import com.squareup.moshi.*;

import java.lang.reflect.Type;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Objects;

import spark.Request;
import spark.Response;
import spark.Route;

public class ViewCSVHandler implements Route {

  private final Server.CSVSearcherContainer searcherContainer;

  /**
   * Constructor searcherContainer some shared state
   * @param searcherContainer the shared searcher object that all endpoints have
   *                          access to
   *                          (note: we *DON'T* want to make a defensive copy here!)
   */
  public ViewCSVHandler(Server.CSVSearcherContainer searcherContainer) {
    this.searcherContainer = searcherContainer;
  }
  /**
   * handle a new request to this endpoint!
   * @param request the request to handle (including query params)
   * @param response use to modify properties of the response
   * @return response content
   * @throws Exception This is part of the interface; we don't have to throw anything.
   */
  @Override
  public Object handle(Request request, Response response) throws Exception {

    // grab request url
    String url = request.url();

    // error check params - we should have exactly 0
    if (request.queryParams().size() != 0) {
      return new ViewCSVResponse(url, null,
        "error_bad_request: Not exactly 0 params").serialize();
    }

    // if searcher is null, it was likely never loaded in the first place
    if (searcherContainer.searcher == null) {
      return new ViewCSVResponse(url, null,"error_datasource: No CSV loaded").serialize();
    }
    return new ViewCSVResponse(url, searcherContainer.searcher.getCSVData(), "success").serialize();
  }

  /**
   * Makes a response map to send, containing details about the request, and
   * our response
   * @param url - the url of the request we received
   * @param response - error/success plus details
   */
  public record ViewCSVResponse(String url, List<List<String>> CSVData, String response) {

    /**
     * @return url, result, and data, serialized as Json
     */
    String serialize() {
      try {
        // add to our response map!
        HashMap<String, Object> map = new HashMap<>();
        map.put("request", url);
        map.put("result", response);
        map.put("data", Objects.requireNonNullElse(CSVData, "null"));

        // .toJson our response map!
        Type responseMap = Types.newParameterizedType(Map.class, String.class, Object.class);
        Moshi moshi = new Moshi.Builder().build();
        JsonAdapter<Map<String, Object>> adapter = moshi.adapter(responseMap);

        return adapter.toJson(map);
      } catch (Exception e) {
        // For debugging purposes, show in the console _why_ this fails
        // Otherwise we'll just get an error 500 from the API in integration
        // testing.
        e.printStackTrace();
        throw e;
      }
    }
  }
}
