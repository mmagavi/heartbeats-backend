package server;

import com.squareup.moshi.JsonAdapter;
import com.squareup.moshi.Moshi;
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.lang.reflect.Type;
import java.net.HttpURLConnection;
import java.net.URL;
import java.util.List;

public class WeatherAPIRequest {

  public static List<String> newRequest(float latitude, float longitude) throws IOException {
    // make our 1st request's URL
    String pointsURL = "https://api.weather.gov/points/" + latitude + "," + longitude;

    // Call the weather API and fetch a pointsResponse object which has a field annotated with
    // @Json(name = "properties") which will fetch the properties field in the JSON.
    // This field is an HourlyForecastURl Object which again has an annotation @Json(name =
    // "forecastHourly"),
    // which will grab the forecastHourly field and store it in the String url field
    WeatherAPIResponse.PointsResponse pointsAPIResponse =
        callWeatherAPI(pointsURL, WeatherAPIResponse.PointsResponse.class);

    // make our 2nd request's URL
    String hourlyForecastURL = pointsAPIResponse.hourlyForecastURL().url();

    // call the weather api and fetch a hourlyForecastResponse object which has a field
    // annotated with @Json(name = "temperature") and @Json(name = "temperatureUnit")
    WeatherAPIResponse.HourlyForecastResponse hourlyForecastResponse =
        callWeatherAPI(hourlyForecastURL, WeatherAPIResponse.HourlyForecastResponse.class);

    // get temperature and unit!
    String temperature =
        hourlyForecastResponse.hourlyforecastPeriods().hourlyForecast().get(0).temp();
    String unit = hourlyForecastResponse.hourlyforecastPeriods().hourlyForecast().get(0).unit();

    return List.of(temperature, unit);
  }

  /**
   * Calls the Weather.gov API, reads in all the JSON into a string and converts it using a moshi
   * adapter which deserializes the JSON string into an object of the specified type.
   *
   * @param url Url for Weather.gov request
   * @param classType class type that the JSON string needs to be deserialized to
   * @param <T> classType
   * @return An object of the type classType
   * @throws IOException In the event the attempt to reach the nwsURL fails
   */
  public static <T> T callWeatherAPI(String url, Type classType) throws IOException {
    // how to make request and read response: https://www.baeldung.com/java-http-request
    URL nwsURL = new URL(url);
    HttpURLConnection urlConnection = (HttpURLConnection) nwsURL.openConnection();
    urlConnection.setRequestMethod("GET");

    // check status
    int responseCode = urlConnection.getResponseCode();
    if (responseCode != 200) {
      // bad request will land us here!
      throw new IOException("error_bad_request: API returned response code: " + responseCode);
    }

    StringBuilder stringBuilder = new StringBuilder();

    try (InputStreamReader inputStreamReader =
            new InputStreamReader(urlConnection.getInputStream());
        BufferedReader bufferedReader = new BufferedReader(inputStreamReader)) {

      String line;
      while ((line = bufferedReader.readLine()) != null) {
        stringBuilder.append(line);
      }

    } catch (IOException ex) {
      throw new IOException(
          "error_bad_json: JSON was not able to be parsed on the API call to /points");
    }

    urlConnection.disconnect();

    Moshi moshi = new Moshi.Builder().build();
    JsonAdapter<T> adapter = moshi.adapter(classType);
    return adapter.fromJson(stringBuilder.toString());
  }
}
