package CSVUtilities.CSVParser;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.Reader;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

/**
 * The Parser class of our project. This is where parsing happens.
 *
 * @author Stephen Rosa - 1/30/23 - Brown University CSCI0320 Spring '23
 * @version 1.0
 */
public class CSVParser<T> {
  private final CreatorFromRow<T> creator;
  private final boolean hasHeaders;
  // an object that implements java.io.Reader, containing CSV data to be parsed
  private final Reader reader;
  public List<String> headers;

  /**
   * constructor for CSVParser class
   *
   * @param creator an object of type CreatorFromRow T, parameterized. comes from
   *     edu.brown.cs.student.main.api.CreatorFromRow
   * @param hasHeaders a boolean value indicating whether the FIRST row of CSV data from CSVData
   *     contains row headers for the data. If this boolean is true, the first row will not be
   *     turned into a T by the parse method's call to creator.create(), as it is not data. Thus, if
   *     you want the first row to be included in the output, do not provide this argument as true.
   * @param reader an object that implements java.io.Reader, containing CSV data to be parsed
   */
  public CSVParser(CreatorFromRow<T> creator, boolean hasHeaders, Reader reader) {
    this.creator = creator;
    this.hasHeaders = hasHeaders;
    this.reader = reader;
    this.headers = null;
  }

  /**
   * the method that parses a reader object representing CSV data. parameterized to return a list of
   * type T, specified by this.creator if this.hasHeaders is true, will not read the first row as
   * data
   *
   * @return a list of T created by this.creator.create()
   * @throws IOException if there is an IO exception with the Reader CSVReader
   * @throws FactoryFailureException if
   */
  public List<T> parse() throws IOException, FactoryFailureException {
    // wrap CSVReader in BufferedReader to get readLine() method
    BufferedReader csvBufferedReader = new BufferedReader(this.reader);
    String line;

    // the list of our parsed CSV rows from CSVReader
    ArrayList<T> parsedList = new ArrayList<>();
    // enter while loop which terminates when line == null (which happens when
    // readLine() reaches EOF)

    if (this.hasHeaders) {
      // get first line out of the way - it is not data
      line = csvBufferedReader.readLine();
      ArrayList<String> row = new ArrayList<String>(Arrays.asList(line.split(",")));
      this.headers = row;
    }

    while ((line = csvBufferedReader.readLine()) != null) { // throws IOException
      // take the newly read line and split it according to a delimiter, ",",
      // separating it into an array of column values, represented as strings.
      // then pass this to the creator
      // split function idea comes from
      // https://javarevisited.blogspot.com/2015/12/how-to-split-comma-separated-string-in-java-example.html#axzz7rqSZLgrb
      ArrayList<String> row = new ArrayList<String>(Arrays.asList(line.split(",")));
      // conversion from array to ArrayList idea comes from
      // https://www.baeldung.com/convert-array-to-list-and-list-to-array
      parsedList.add(this.creator.create(row)); // throws FactoryFailureException
    }
    // at this point, line == null, meaning we have read every line we can from
    // this BufferedReader
    return parsedList;
  }
}
